6   package io.github.olyutorskii.doubdabc;
7   
8   /**
9    * BCD array utilities but experimental.
10   */
11  public final class BcdArrays {
12  
13      private static final int INT_SLOTS =
14              Integer.SIZE / BcdUtils.BCD_BITSIZE;
15  
16      private static final int MSB_INTMASK = 0b1  << (Integer.SIZE - 1);
17      private static final int LSB_INTMASK = 0b1;
18  
19      private static final int BCD_MASK =
20              (0b1 << BcdUtils.BCD_BITSIZE) - 1;
21      private static final int ARABICUC_MASK = 0b0011_0000;
22  
23  
24      /**
25       * Hidden constructor.
26       */
27      private BcdArrays(){
28          assert false;
29      }
30  
31  
32      /**
33       * Convert int value to Arabic-decimal character sequence.
34       *
35       * <p>Output nothing if value 0.
36       *
37       * <p>sign-bit is treated as unsigned.
38       * There is no minus-sign output.
39       *
40       * <p>DDA implementations without any instance field.
41       *
42       * @param iVal int value
43       * @param cbuf char output
44       * @param offset output start offset
45       * @return output length
46       */
47      public static int int2ArabicArray(final int iVal,
48                                          final char[] cbuf, final int offset){
49          int iHigh = 0b0;
50          int iLow  = 0b0;
51  
52          boolean skipLeading = true;
53          boolean noHalfCarry = true;
54          int sig = 0;
55  
56          for(int bitCt = 0; bitCt < Integer.SIZE; bitCt++){
57              final int scanMask = MSB_INTMASK >>> bitCt;
58              final int bitAppear = iVal & scanMask;
59  
60              if(skipLeading){
61                  if(bitAppear == 0b0) continue;
62                  skipLeading = false;
63              }
64              sig++;
65  
66              int bqVal;
67              int shiftedBcd;
68  
69              bqVal = BcdUtils.toBiQuinary(iLow);
70              shiftedBcd = bqVal << 1;
71  
72              if(bitAppear == 0b0){
73                  iLow = shiftedBcd;
74              }else{
75                  iLow = shiftedBcd | LSB_INTMASK;
76              }
77  
78              // through iHigh
79              // if iLow has 8-bcd "67108863"(2^26-1) or lower
80              if(sig < 27) continue;
81  
82              int halfCarry = bqVal & MSB_INTMASK;
83  
84              if(noHalfCarry){
85                  if(halfCarry == 0b0) continue;
86                  noHalfCarry = false;
87              }
88  
89              bqVal = BcdUtils.toBiQuinary(iHigh);
90              shiftedBcd = bqVal << 1;
91  
92              if(halfCarry == 0b0){
93                  iHigh = shiftedBcd;
94              }else{
95                  iHigh = shiftedBcd | LSB_INTMASK;
96              }
97          }
98  
99          int cidx = offset;
100         final int clzLow;
101 
102         if(iHigh == 0b0){
103             clzLow = BcdUtils.clzNibble(iLow);
104         }else{
105             clzLow = 0;
106 
107             final int clzHigh = BcdUtils.clzNibble(iHigh);
108 
109             for(int bcdCt = clzHigh; bcdCt < INT_SLOTS; bcdCt++){
110                 final int nibble = (iHigh >>> ((7 - bcdCt) << 2)) & BCD_MASK;
111                 final char decimalCh = (char)(nibble | ARABICUC_MASK);
112                 cbuf[cidx++] = decimalCh;
113             }
114         }
115 
116         for(int bcdCt = clzLow; bcdCt < INT_SLOTS; bcdCt++){
117             final int nibble = (iLow >>> ((7 - bcdCt) << 2)) & BCD_MASK;
118             final char decimalCh = (char)(nibble | ARABICUC_MASK);
119             cbuf[cidx++] = decimalCh;
120         }
121 
122         return cidx - offset;
123     }
124 
125 }
