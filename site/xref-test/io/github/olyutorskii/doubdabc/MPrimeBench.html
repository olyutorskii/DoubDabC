4   package io.github.olyutorskii.doubdabc;
5   
6   import java.math.BigInteger;
7   
8   /**
9    * Mersenne Prime #32 to decimal converting benchmark test.
10   *
11   * It takes about 120sec on (Core i5 2.6GHz + JDK1.8(64bit))
12   * @see <a target="_blank"
13   * href="https://en.wikipedia.org/wiki/Mersenne_prime">Mersenne prime</a>
14   */
15  public class MPrimeBench {
16  
17      static final int PWR = 756839;   // Mersenne Prime #32 is 2^756839 -1
18      static final int COLS = 227832;  // *log10(2) +1
19  
20      public static void main(String[] args){
21          System.out.print("Mersenne Prime #32 (2^756839 -1) = ");
22          dumpM32();
23          return;
24      }
25  
26      static void dumpM32(){
27          BcdRegister br = new BcdRegister(COLS);
28          BcdSequence bs = new BcdSequence(br);
29          for(int ct = 0; ct < PWR; ct++){
30              br.pushLsb(1);
31          }
32          System.out.println(bs);
33          return;
34      }
35  
36      //  fast but spawning garbages hell, maybe
37      static void dumpM32_BigInt(){
38          BigInteger twopow = BigInteger.ONE.shiftLeft(PWR);
39          BigInteger mprime = twopow.subtract(BigInteger.ONE);
40          System.out.println(mprime);
41          return;
42      }
43  
44  }
