40      /**
41       * Test of int2ArabicArray method, of class BcdArrays.
42       */
43      @Test
44      public void testInt2ArabicArray() {
45          System.out.println("int2ArabicArray");
46  
47          int result;
48          char[] buf;
49  
50          buf = new char[100];
51  
52          result = BcdArrays.int2ArabicArray(1, buf, 0);
53          assertEquals(result, 1);
54          assertEquals("1", new String(buf, 0, result));
55  
56          result = BcdArrays.int2ArabicArray(1234567890, buf, 0);
57          assertEquals(result, 10);
58          assertEquals("1234567890", new String(buf, 0, result));
59  
60          result = BcdArrays.int2ArabicArray(99999999, buf, 0);
61          assertEquals(result, 8);
62          assertEquals("99999999", new String(buf, 0, result));
63  
64          result = BcdArrays.int2ArabicArray(100000000, buf, 0);
65          assertEquals(result, 9);
66          assertEquals("100000000", new String(buf, 0, result));
67  
68          result = BcdArrays.int2ArabicArray(67_108_863, buf, 0);
69          assertEquals(result, 8);
70          assertEquals("67108863", new String(buf, 0, result));
71  
72          result = BcdArrays.int2ArabicArray(67_108_864, buf, 0);
73          assertEquals(result, 8);
74          assertEquals("67108864", new String(buf, 0, result));
75  
76          result = BcdArrays.int2ArabicArray(0, buf, 0);
77          assertEquals(result, 0);
78          assertEquals("", new String(buf, 0, result));
79  
80          result = BcdArrays.int2ArabicArray(-1, buf, 0);
81          assertEquals(result, 10);
82          assertEquals("4294967295", new String(buf, 0, result));
83  
84          result = BcdArrays.int2ArabicArray(Integer.MAX_VALUE, buf, 0);
85          assertEquals(result, 10);
86          assertEquals("2147483647", new String(buf, 0, result));
87  
88          result = BcdArrays.int2ArabicArray(Integer.MIN_VALUE, buf, 0);
89          assertEquals(result, 10);
90          assertEquals("2147483648", new String(buf, 0, result));
91  
92          result = BcdArrays.int2ArabicArray(234, buf, 0);
93          assertEquals(result, 3);
94          assertEquals('2', buf[0]);
95          assertEquals('3', buf[1]);
96          assertEquals('4', buf[2]);
97  
98          buf[0] = 'A';
99          buf[1] = 'B';
100         buf[2] = 'C';
101         result = BcdArrays.int2ArabicArray(234, buf, 2);
102         assertEquals(result, 3);
103         assertEquals("AB234", new String(buf, 0, 2+result));
104 
105         return;
106     }
